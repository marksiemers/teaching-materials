Step by Step:

  • Create an array of every link on the page using document.querySelectorAll(cssSelector)
  • Iterate through the array. In each iteration of the loop:
    • Find the current href using element.getAttribute(name), and store into a variable
    • Generate a thumbnail URL using youtube.generateThumbnailUrl(videoUrl)
    • Create an IMG element using document.createElement(tagName)
    • Set the "src" of the IMG element using element.setAttribute(name, value)
    • Append the IMG to the link using element.appendChild(element)

    <script type="text/javascript" src="https://dbc-gdi.netlify.com/common/youtube.js"></script>

    <script>
    var addVideoThumbnail = function($videoLink) {
      // What jQuery method can we use to get the linkUrl?
      var linkUrl = $videoLink.attr('href');

      // We need a thumbnail URL from youtube, how do we get it?
      // See https://dbc-gdi.netlify.com/jquery2/youtube_doc.html
      var thumbnailUrl = youtube.generateThumbnailUrl(linkUrl);

      // We need a new image element, with the src set to the thumbnailUrl
      // How can we do this using jQuery?
      var $thumbnailImg = $('<img>');
      $thumbnailImg.attr('src', thumbnailUrl);

      // We need to append the thumbnail to the video link.
      // How do we use jQuery to do that?
      $videoLink.append(thumbnailImg);
    };

    // Use jQuery to get all the links
    var $videoLinks = $('a');

    // How do we add a thumbnail to all the videos within this for loop?
    for (var i = 0; i < $videoLinks.length; i++) {
      var $videoLink = $($videoLinks[i]);
      addVideoThumbnail($videoLink);
    }
    </script>
  </body>
